    <div id="raw-container">Loading file...</div>

    <script>
        // Change 'yourfile.txt' to the actual name of your file
        const fileName = 'yourfile.txt';

        async function loadTxt() {
            try {
                const response = await fetch(fileName);
                if (!response.ok) throw new Error('Could not find the file');
                
                const data = await response.text();
                
                // .textContent is vitalâ€”it prevents the browser from 
                // trying to parse symbols inside the txt as HTML.
                document.getElementById('raw-container').textContent = data;
            } catch (err) {
                document.getElementById('raw-container').textContent = "Error: " + err.message;
            }
        }

        loadTxt();
    </script>
